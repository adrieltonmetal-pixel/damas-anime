import { initializeApp } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-app.js"; import { getDatabase, ref, set } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-database.js"; const firebaseConfig = { apiKey: "SUA_CHAVE", authDomain: "SEU_DOMINIO", projectId: "SEU_ID", databaseURL: "SUA_DATABASE_URL", storageBucket: "SEU_BUCKET", messagingSenderId: "SEU_ID", appId: "SEU_APP_ID" }; const app = initializeApp(firebaseConfig); const db = getDatabase(app); function criarSala() { const codigo = Math.random().toString(36).substring(2, 7); console.log("Tentando salvar no Firebase..."); set(ref(db, 'salas/' + codigo), { criadaEm: Date.now() }) .then(() => { console.log("SALVO COM SUCESSO"); alert("Sala: " + codigo); }) .catch((error) => { console.error("ERRO:", error); }); } window.criarSala = criarSala;
